25       protected void onCreate(Bundle savedInstanceState) { 
26           super.onCreate(savedInstanceState); 
27           setContentView(R.layout.activity_game_over); 
28    
29           TextView endGameMessage = (TextView) findViewById(R.id.endGameMessage); 
30           TextView scoreText = (TextView) findViewById(R.id.gameOverScore); 
31           TextView longestStreakText = (TextView) findViewById(R.id.longestStreak); 
32           TextView levelText = (TextView) findViewById(R.id.level); 
33    
34            SharedPreferences sharedPref = getSharedPreferences("userData", Context.MODE_PRIVATE); 
35            SharedPreferences.Editor editor = sharedPref.edit(); 
36    
37           //editor.putBoolean("isFirstGame", false); 
38    
39           Bundle scoreData = getIntent().getExtras(); 
40           finalScore = scoreData.getInt("score"); 
41    
42           Bundle streakData = getIntent().getExtras(); 
43           longestStreak = streakData.getInt("streak"); 
44    
45           Bundle diffData = getIntent().getExtras(); 
46           diffNum = diffData.getInt("difficulty"); 
47    
48           if (scoreData == null) { 
49               return; 
50           } 
51           else 
52           { 
53               if (currentHighScore == -1){ 
54                   currentHighScore = finalScore; 
55                   editor.putInt("highScore", currentHighScore); 
56                   editor.apply(); 
57                   endGameMessage.setText("New High Score!"); 
58               } 
59               else { 
60                   if (finalScore > sharedPref.getInt("highScore",currentHighScore)) { 
61                       currentHighScore = finalScore; 
62                       editor.putInt("highScore", currentHighScore); 
63                       editor.apply(); 
64                       endGameMessage.setText("New High Score!"); 
65                   } else { 
66                       endGameMessage.setText("Good game!"); 
67                   } 
68    
69               } 
70    
71               scoreText.setText("Score: " + finalScore); 
72               longestStreakText.setText("Longest Streak: " + longestStreak); 
73               levelText.setText("Level: "); 
74           } 
75       }
